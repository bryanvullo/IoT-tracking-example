// terminal commands to start micro 
    // first run this to get to the correct directory
cd Documents/GitHub/snowplow

    // then run this command to run docker
docker run \
  --mount type=bind,source=$(pwd)/Micro,destination=/config \
  -p 9090:9090 \
  snowplow/snowplow-micro:1.3.1 \
  --collector-config /config/micro.conf \
  --iglu /config/iglu.json
